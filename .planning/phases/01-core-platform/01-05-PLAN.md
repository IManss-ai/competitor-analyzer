  <action>Build the approval queue route and action card templates</action>
  <detail>
    Create app/routes/queue.py:

    ```python
    from fastapi import APIRouter, Request, Form, Depends
    from fastapi.responses import HTMLResponse
    from fastapi.templating import Jinja2Templates
    from sqlalchemy import select
    from app.db import get_session
    from app.models import ApprovedAction, ChangeEvent, Competitor
    from app.session import require_current_user
    from datetime import datetime, timezone

    router = APIRouter(prefix="/queue")
    templates = Jinja2Templates(directory="templates")

    @router.get("/", response_class=HTMLResponse)
    async def approval_queue(request: Request, db=Depends(get_session), user_id=Depends(require_current_user)):
        # Get pending actions (approved_at is null)
        pending = db.execute(
            select(ApprovedAction, ChangeEvent, Competitor)
            .join(ChangeEvent, ApprovedAction.change_event_id == ChangeEvent.id)
            .join(Competitor, ChangeEvent.competitor_id == Competitor.id)
            .where(ApprovedAction.user_id == user_id)
            .where(ApprovedAction.approved_at == None)
            .order_by(ApprovedAction.created_at.desc())
        ).all()
        return templates.TemplateResponse("queue.html", {
            "request": request,
            "pending_actions": pending,
        })

    @router.post("/{action_id}/approve", response_class=HTMLResponse)
    async def approve_action(
        request: Request, action_id: str,
        edited_text: str = Form(None),
        db=Depends(get_session), user_id=Depends(require_current_user)
    ):
        """
        Approve an action draft. If edited_text differs from original_draft,
        store it as the edited version (brand voice capture).
        """
        action = db.get(ApprovedAction, action_id)
        if not action or str(action.user_id) != user_id:
            return HTMLResponse("Not found", status_code=404)

        action.approved_at = datetime.now(timezone.utc)
        # Brand voice capture: only store edited_text if founder actually changed it
        if edited_text and edited_text.strip() != action.original_draft.strip():
            action.edited_text = edited_text.strip()
        db.commit()

        # HTMX: return empty response to remove the card from queue
        return HTMLResponse("")

    @router.post("/{action_id}/dismiss", response_class=HTMLResponse)
    async def dismiss_action(
        request: Request, action_id: str,
        db=Depends(get_session), user_id=Depends(require_current_user)
    ):
        """Dismiss without approving — removes from queue, no brand voice capture."""
        action = db.get(ApprovedAction, action_id)
        if action and str(action.user_id) == user_id:
            action.approved_at = datetime.now(timezone.utc)
            db.commit()
        return HTMLResponse("")
    ```

## Success Criteria

- Approval queue shows pending action cards with editable textarea and copy button

- Dismiss removes card without brand voice capture

## must_haves

- Copy-to-clipboard uses the textarea value (not original_draft) so founder copies their edited version
- Brand voice comparison: edited_text stored ONLY if it differs from original_draft (not on every approve)
- HTMX targets remove cards without full-page reload — the approved card disappears immediately
- Pending filter is approved_at == None (null), not a separate status column
